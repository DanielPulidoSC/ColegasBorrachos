/*  Sentències de creació de taules */
create table tipus
(nomT char(50), 
primary key (nomT));

create table població
(nomP char(50), 
barri char(50), 
primary key (nomP,barri));

create table bar(
nom char(50),
carrer char(50), 
preu_birra float, 
nomP char(50), 
barri char(50),
nomT char(50), 
primary key (nom,nomP,barri,nomT),
foreign key (nomP,barri) references població,
foreign key (nomT) references tipus);
